      def get_character( input = STDIN )
        @stdin_handle ||= GetStdHandle(STD_INPUT_HANDLE)
        
        begin
            SetConsoleEcho(@stdin_handle, false)
            input.getbyte
        ensure
            SetConsoleEcho(@stdin_handle, true)
        end        
      end
