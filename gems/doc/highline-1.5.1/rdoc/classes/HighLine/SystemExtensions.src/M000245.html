      def GetConsoleMode( console_handle )
        @@apiGetConsoleMode ||= Win32API.new( "kernel32", "GetConsoleMode",
                                              ['L', 'P'], 'I' )
        
        mode = ' ' * 4
        @@apiGetConsoleMode.call(console_handle, mode)
        mode.unpack('L')[0]
      end
