      def SetConsoleEcho( console_handle, on )
        mode = GetConsoleMode(console_handle)
        
        # toggle the console echo bit
        if on
            mode |=  ENABLE_ECHO_INPUT
        else
            mode &= ~ENABLE_ECHO_INPUT
        end
        
        ok = SetConsoleMode(console_handle, mode)    
      end
